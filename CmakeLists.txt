cmake_minimum_required(VERSION 3.10)

project(CDGeometry)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED on)

include_directories("${CMAKE_CURRENT_SOURCE_DIR}/third/include")
link_libraries("${CMAKE_CURRENT_SOURCE_DIR}/third/lib")

file(GLOB_RECURSE SRC_FILES  "${CMAKE_CURRENT_SOURCE_DIR}/*.cpp" "${CMAKE_CURRENT_SOURCE_DIR}/*.h" "${CMAKE_CURRENT_SOURCE_DIR}/*.c")
list(FILTER SRC_FILES EXCLUDE REGEX ".*CMakeFiles.*")

add_executable(CDGeometry ${SRC_FILES})

target_link_libraries(CDGeometry ${CMAKE_CURRENT_SOURCE_DIR}/third/lib/glfw3.lib)
